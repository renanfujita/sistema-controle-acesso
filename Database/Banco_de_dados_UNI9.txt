CREATE DATABASE SistemaAcesso;
GO

USE SistemaAcesso;
GO

CREATE TABLE TiposUsuario (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nome VARCHAR(30) NOT NULL UNIQUE
);
GO

GO

CREATE TABLE Usuarios (
    Id BIGINT IDENTITY(1,1) PRIMARY KEY,
    Username VARCHAR(50) NOT NULL UNIQUE,
    Senha VARCHAR(255) NOT NULL,
    Nome VARCHAR(150) NOT NULL,
    TipoId INT NOT NULL,
    CriadoEm DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
    AtualizadoEm DATETIME2 NOT NULL DEFAULT SYSDATETIME(),

    CONSTRAINT FK_Usuarios_TiposUsuario
        FOREIGN KEY (TipoId)
        REFERENCES TiposUsuario(Id)
);
GO

CREATE INDEX IX_Usuarios_Nome
ON Usuarios(Nome);
GO

CREATE INDEX IX_Usuarios_Username
ON Usuarios(Username);
GO

CREATE TABLE Acessos (
    Id BIGINT IDENTITY(1,1) PRIMARY KEY,
    UsuarioId BIGINT NOT NULL,
    DataHora DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
    IP VARCHAR(45) NULL,
    Navegador VARCHAR(255) NULL,

    CONSTRAINT FK_Acessos_Usuarios
        FOREIGN KEY (UsuarioId)
        REFERENCES Usuarios(Id)
        ON DELETE CASCADE
);
GO

CREATE INDEX IX_Acessos_UsuarioId
ON Acessos(UsuarioId);
GO

CREATE INDEX IX_Acessos_DataHora
ON Acessos(DataHora DESC);
GO

CREATE TRIGGER TR_Usuarios_AtualizadoEm
ON Usuarios
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE u
    SET AtualizadoEm = SYSDATETIME()
    FROM Usuarios u
    INNER JOIN inserted i
        ON u.Id = i.Id;
END;
GO

CREATE VIEW VW_Usuarios AS
SELECT
    U.Id,
    U.Nome,
    U.Username,
    T.Nome AS Tipo,
    U.CriadoEm,
    U.AtualizadoEm
FROM Usuarios U
INNER JOIN TiposUsuario T
    ON T.Id = U.TipoId;
GO

CREATE VIEW VW_HistoricoAcessos AS
SELECT
    A.Id,
    U.Nome,
    U.Username,
    A.DataHora,
    A.IP,
    A.Navegador
FROM Acessos A
INNER JOIN Usuarios U
    ON U.Id = A.UsuarioId;
GO

CREATE PROCEDURE SP_Login
(
    @Username VARCHAR(50),
    @Senha VARCHAR(255)
)
AS
BEGIN
    SELECT
        U.Id,
        U.Nome,
        U.Username,
        T.Nome AS Tipo
    FROM Usuarios U
    INNER JOIN TiposUsuario T
        ON T.Id = U.TipoId
    WHERE U.Username = @Username
    AND U.Senha = @Senha;
END;
GO

CREATE PROCEDURE SP_RegistrarAcesso
(
    @UsuarioId BIGINT,
    @IP VARCHAR(45) = NULL,
    @Navegador VARCHAR(255) = NULL
)
AS
BEGIN
    INSERT INTO Acessos
    (
        UsuarioId,
        IP,
        Navegador
    )
    VALUES
    (
        @UsuarioId,
        @IP,
        @Navegador
    );
END;
GO

CREATE PROCEDURE SP_PromoverAdministrador
(
    @UsuarioId BIGINT
)
AS
BEGIN
    UPDATE Usuarios
    SET TipoId = 1
    WHERE Id = @UsuarioId;
END;
GO
